### 9.3 AI配置持久化

```javascript
// stores/ai.js
export const useAIStore = defineStore('ai', () => {
  const llmConfig = ref({
    provider: 'openai',
    model: 'gpt-4o-mini',
    apiKey: '',
    baseUrl: ''
  })

  const embeddingConfig = ref({
    provider: 'openai',
    model: 'text-embedding-3-small',
    apiKey: '',
    dimension: 1536
  })

  // 保存到localStorage
  function saveConfig() {
    localStorage.setItem('ai_config', JSON.stringify({
      llm: llmConfig.value,
      embedding: embeddingConfig.value
    }))
  }

  // 从localStorage加载
  function loadConfig() {
    const saved = localStorage.getItem('ai_config')
    if (saved) {
      const config = JSON.parse(saved)
      llmConfig.value = config.llm
      embeddingConfig.value = config.embedding
    }
  }

  return {
    llmConfig,
    embeddingConfig,
    saveConfig,
    loadConfig
  }
})
```
