## 方向一：RAG 知识库与知识图谱

### 1.1 目标

将知识库从"文档存储"升级为"智能知识库"：文档上传后自动解析、分块、向量化，支持 RAG 语义检索，为 AI 工作台提供真实上下文。

### 1.2 现状

| 已有 | 缺失 |
|------|------|
| Document 模型（name, file_path, ai_summary, module_ids） | 文档解析服务（PDF/Word → 纯文本） |
| Module 模型（AI 标签字典） | 文本分块服务（Chunk） |
| FeaturePoint 模型（AI 提取的功能点） | 向量化存储（ChromaDB） |
| Graph/GraphNode/GraphEdge（图谱结构表） | 真实的图数据库（Neo4j）或图算法 |
| AI Provider 配置（api_key, model, endpoint_url） | RAG 检索服务 |

### 1.3 技术选型

| 组件 | 选型 | 理由 |
|------|------|------|
| 文档解析 | `pdfplumber` + `python-docx` + `openpyxl` | 纯 Python，无外部依赖 |
| 向量数据库 | **ChromaDB**（嵌入式优先） | 轻量、Python 原生、Docker 部署简单 |
| Embedding 模型 | OpenAI `text-embedding-3-small`（远程）/ `BGE-M3`（本地） | 通过 AIProvider 配置切换 |
| 图数据库 | **暂用 PostgreSQL**（已有 Graph 表）→ 远期可选 Neo4j | 当前规模 PG 够用，避免引入新组件 |

```python
class VectorStore:
    def __init__(self, chroma_host="chromadb", chroma_port=8000):
        self.client = chromadb.HttpClient(host=chroma_host, port=chroma_port)

    def index_chunks(self, project_id: int, chunks: list[dict], embeddings: list[list[float]]):

        results = collection.query(query_embeddings=[query_embedding], n_results=top_k, where=filter)
```

```python
# llm_adapter.py
class LLMAdapter:
    """统一 LLM 调用入口"""

    def __init__(self, db: Session):
        self.db = db

    def call(self, task_type: str, prompt: str, **kwargs) -> dict:
        """
        根据 task_type 查询 ModelStrategy → 获取 Provider → 调用
        自动记录 AICallLog（tokens, 耗时, 状态）
        """
        strategy = self._get_strategy(task_type)
        provider = self._get_provider(strategy.provider_id)

        # 记录开始时间
        start = time.time()
        try:
            result = self._dispatch(provider, prompt, strategy, **kwargs)
            duration = int((time.time() - start) * 1000)

            # 记录成功日志
            self._log_call(provider, task_type, result, duration, "成功")
            return result
        except Exception as e:
            duration = int((time.time() - start) * 1000)
            self._log_call(provider, task_type, {"error": str(e)}, duration, "失败")
            raise

    def _dispatch(self, provider, prompt, strategy, **kwargs):
        """路由到对应服务商"""
        if provider.provider_type == "OpenAI":
            return OpenAIProvider().call(provider, prompt, strategy, **kwargs)
        elif provider.provider_type == "Anthropic":
            return AnthropicProvider().call(provider, prompt, strategy, **kwargs)
        elif provider.provider_type == "DeepSeek":
            return DeepSeekProvider().call(provider, prompt, strategy, **kwargs)
        else:
            return CustomProvider().call(provider, prompt, strategy, **kwargs)
```

```python
# llm_providers.py
class OpenAIProvider:
    def call(self, provider, prompt, strategy, **kwargs):
        import httpx
        resp = httpx.post(
            f"{provider.endpoint_url or 'https://api.openai.com'}/v1/chat/completions",
            headers={"Authorization": f"Bearer {provider.api_key}"},
            json={
                "model": strategy.model_name,
                "messages": [{"role": "user", "content": prompt}],
                "temperature": strategy.temperature,
                "max_tokens": provider.max_tokens,
            },
            timeout=120,
        )
        data = resp.json()
        return {
            "content": data["choices"][0]["message"]["content"],
            "input_tokens": data["usage"]["prompt_tokens"],
            "output_tokens": data["usage"]["completion_tokens"],
            "model": strategy.model_name,
        }

class AnthropicProvider:
    def call(self, provider, prompt, strategy, **kwargs):
        import httpx
        resp = httpx.post(
            "https://api.anthropic.com/v1/messages",
            headers={
                "x-api-key": provider.api_key,
                "anthropic-version": "2023-06-01",
                "content-type": "application/json",
            },
            json={
                "model": strategy.model_name,
                "max_tokens": provider.max_tokens,
                "messages": [{"role": "user", "content": prompt}],
            },
            timeout=120,
        )
        data = resp.json()
        return {
            "content": data["content"][0]["text"],
            "input_tokens": data["usage"]["input_tokens"],
            "output_tokens": data["usage"]["output_tokens"],
            "model": strategy.model_name,
        }

# DeepSeek 兼容 OpenAI 格式，endpoint_url 不同
class DeepSeekProvider(OpenAIProvider):
    def call(self, provider, prompt, strategy, **kwargs):
        provider.endpoint_url = provider.endpoint_url or "https://api.deepseek.com"
        return super().call(provider, prompt, strategy, **kwargs)
```

```python
# pipeline_executor.py
class PipelineExecutor:
    def __init__(self, db: Session):
        self.db = db
        self.llm = LLMAdapter(db)

    async def execute(self, execution: PipelineExecution):
        """真实执行流水线"""
        execution.status = "running"
        execution.started_at = datetime.utcnow()

        for stage in execution.stages:
            stage.status = "running"
            stage.started_at = datetime.utcnow()
            self.db.flush()

            try:
                # 根据 stage_no 执行对应逻辑
                result = await self._execute_stage(stage, execution)
                stage.status = "completed"
                stage.result_summary = result.get("summary", {})
                stage.model = result.get("model", "")
                stage.input_tokens = result.get("input_tokens", 0)
                stage.output_tokens = result.get("output_tokens", 0)
            except Exception as e:
                stage.status = "failed"
                stage.result_summary = {"error": str(e)}
                execution.status = "failed"
                break

        if execution.status != "failed":
            execution.status = "completed"
        execution.completed_at = datetime.utcnow()
        self.db.commit()
```

```python
            # 支持通配符: "testcase:*" 匹配 "testcase:view"
            has_permission = any(
                p == permission or p.endswith(":*") and permission.startswith(p[:-1])
                for p in permissions
            )

            if not has_permission:
                raise HTTPException(status_code=403, detail=f"权限不足: {permission}")

            return await func(*args, current_user=current_user, **kwargs)
        return wrapper
    return decorator


# 使用示例

    ...
```

```javascript
// permissions.js
const userPermissions = ref([])

export function loadPermissions() {
  // 从 /auth/me 获取用户角色权限
  const user = store.state.user
  if (user?.role?.permissions) {
    userPermissions.value = user.role.permissions
  }
}

export function hasPermission(permission) {
  if (userPermissions.value.includes('*')) return true
  const [module] = permission.split(':')
  return userPermissions.value.includes(permission) ||
         userPermissions.value.includes(`${module}:*`)
}
```
